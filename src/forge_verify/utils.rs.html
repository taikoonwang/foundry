32/// Enum to represent the type of bytecode being verified
33#[derive(Debug, Serialize, Deserialize, Clone, Copy, ValueEnum)]
34pub enum BytecodeType {
35    #[serde(rename = "creation")]
36    Creation,
37    #[serde(rename = "runtime")]
38    Runtime,
39}
40
41impl BytecodeType {
42    /// Check if the bytecode type is creation
43    pub fn is_creation(&self) -> bool {
44        matches!(self, Self::Creation)
45    }
46
47    /// Check if the bytecode type is runtime
48    pub fn is_runtime(&self) -> bool {
49        matches!(self, Self::Runtime)
50    }
51}

54pub struct JsonResult {

61pub fn match_bytecodes(
62    local_bytecode: &[u8],
63    bytecode: &[u8],
64    constructor_args: &[u8],
65    is_runtime: bool,
66    bytecode_hash: BytecodeHash,
67) -> Option<VerificationType> {
68    // 1. Try full match
69    if local_bytecode == bytecode {
70        // If the bytecode_hash = 'none' in Config. Then it's always a partial match according to
71        // sourcify definitions. Ref: https://docs.sourcify.dev/docs/full-vs-partial-match/.
72        if bytecode_hash == BytecodeHash::None {
73            return Some(VerificationType::Partial);
74        }
75
76        Some(VerificationType::Full)
77    } else {
78        is_partial_match(local_bytecode, bytecode, constructor_args, is_runtime)
79            .then_some(VerificationType::Partial)
80    }
81}

83pub fn build_project(
84    args: &VerifyBytecodeArgs,
85    config: &Config,
86) -> Result<CompactContractBytecode> {
87    let project = config.project()?;
88    let compiler = ProjectCompiler::new();
89
90    let mut output = compiler.compile(&project)?;
91
92    let artifact = output
93        .remove_contract(&args.contract)
94        .ok_or_eyre("Build Error: Contract artifact not found locally")?;
95
96    Ok(artifact.into_contract_bytecode())
97}

99pub fn build_using_cache(
100    args: &VerifyBytecodeArgs,
101    etherscan_settings: &Metadata,
102    config: &Config,
103) -> Result<CompactContractBytecode> {
104    let project = config.project()?;
105    let cache = project.read_cache_file()?;
106    let cached_artifacts = cache.read_artifacts::<CompactContractBytecode>()?;
107
108    for (key, value) in cached_artifacts {
109        let name = args.contract.name.to_owned() + ".sol";
110        let version = etherscan_settings.compiler_version.to_owned();
111        // Ignores vyper
112        if version.starts_with("vyper:") {
113            eyre::bail!("Vyper contracts are not supported")
114        }
115        // Parse etherscan version string
116        let version = version.split('+').next().unwrap_or("").trim_start_matches('v').to_string();
117
118        // Check if `out/directory` name matches the contract name
119        if key.ends_with(name.as_str()) {
120            let name = name.replace(".sol", ".json");
121            for artifact in value.into_values().flatten() {
122                // Check if ABI file matches the name
123                if !artifact.file.ends_with(&name) {
124                    continue;
125                }
126
127                // Check if Solidity version matches
128                if let Ok(version) = Version::parse(&version) {
129                    if !(artifact.version.major == version.major &&
130                        artifact.version.minor == version.minor &&
131                        artifact.version.patch == version.patch)
132                    {
133                        continue;
134                    }
135                }
136
137                return Ok(artifact.artifact)
138            }
139        }
140    }
141
142    eyre::bail!("couldn't find cached artifact for contract {}", args.contract.name)
143}

145pub fn print_result(
146    res: Option<VerificationType>,
147    bytecode_type: BytecodeType,
148    json_results: &mut Vec<JsonResult>,
149    etherscan_config: &Metadata,
150    config: &Config,
151) {
152    if let Some(res) = res {
153        if !shell::is_json() {
154            let _ = sh_println!(
155                "{} with status {}",
156                format!("{bytecode_type:?} code matched").green().bold(),
157                res.green().bold()
158            );
159        } else {
160            let json_res = JsonResult { bytecode_type, match_type: Some(res), message: None };
161            json_results.push(json_res);
162        }
163    } else if !shell::is_json() {
164        let _ = sh_err!(
165            "{bytecode_type:?} code did not match - this may be due to varying compiler settings"
166        );
167        let mismatches = find_mismatch_in_settings(etherscan_config, config);
168        for mismatch in mismatches {
169            let _ = sh_eprintln!("{}", mismatch.red().bold());
170        }
171    } else {
172        let json_res = JsonResult {
173            bytecode_type,
174            match_type: res,
175            message: Some(format!(
176                "{bytecode_type:?} code did not match - this may be due to varying compiler settings"
177            )),
178        };
179        json_results.push(json_res);
180    }
181}

183fn is_partial_match(
184    mut local_bytecode: &[u8],
185    mut bytecode: &[u8],
186    constructor_args: &[u8],
187    is_runtime: bool,
188) -> bool {
189    // 1. Check length of constructor args
190    if constructor_args.is_empty() || is_runtime {
191        // Assume metadata is at the end of the bytecode
192        return try_extract_and_compare_bytecode(local_bytecode, bytecode)
193    }
194
195    // If not runtime, extract constructor args from the end of the bytecode
196    bytecode = &bytecode[..bytecode.len() - constructor_args.len()];
197    local_bytecode = &local_bytecode[..local_bytecode.len() - constructor_args.len()];
198
199    try_extract_and_compare_bytecode(local_bytecode, bytecode)
200}
201
202fn try_extract_and_compare_bytecode(mut local_bytecode: &[u8], mut bytecode: &[u8]) -> bool {
203    local_bytecode = ignore_metadata_hash(local_bytecode);
204    bytecode = ignore_metadata_hash(bytecode);
205
206    // Now compare the local code and bytecode
207    local_bytecode == bytecode
208}

210fn find_mismatch_in_settings(
211    etherscan_settings: &Metadata,
212    local_settings: &Config,
213) -> Vec<String> {
214    let mut mismatches: Vec<String> = vec![];
215    if etherscan_settings.evm_version != local_settings.evm_version.to_string().to_lowercase() {
216        let str = format!(
217            "EVM version mismatch: local={}, onchain={}",
218            local_settings.evm_version, etherscan_settings.evm_version
219        );
220        mismatches.push(str);
221    }
222    let local_optimizer: u64 = if local_settings.optimizer == Some(true) { 1 } else { 0 };
223    if etherscan_settings.optimization_used != local_optimizer {
224        let str = format!(
225            "Optimizer mismatch: local={}, onchain={}",
226            local_settings.optimizer.unwrap_or(false),
227            etherscan_settings.optimization_used
228        );
229        mismatches.push(str);
230    }
231    if local_settings.optimizer_runs.is_some_and(|runs| etherscan_settings.runs != runs as u64) ||
232        (local_settings.optimizer_runs.is_none() && etherscan_settings.runs > 0)
233    {
234        let str = format!(
235            "Optimizer runs mismatch: local={}, onchain={}",
236            local_settings.optimizer_runs.unwrap(),
237            etherscan_settings.runs
238        );
239        mismatches.push(str);
240    }
241
242    mismatches
243}

245pub fn maybe_predeploy_contract(
246    creation_data: Result<ContractCreationData, EtherscanError>,
247) -> Result<(Option<ContractCreationData>, bool), eyre::ErrReport> {
248    let mut maybe_predeploy = false;
249    match creation_data {
250        Ok(creation_data) => Ok((Some(creation_data), maybe_predeploy)),
251        // Ref: https://explorer.mode.network/api?module=contract&action=getcontractcreation&contractaddresses=0xC0d3c0d3c0D3c0d3C0D3c0D3C0d3C0D3C0D30010
252        Err(EtherscanError::EmptyResult { status, message })
253            if status == "1" && message == "OK" =>
254        {
255            maybe_predeploy = true;
256            Ok((None, maybe_predeploy))
257        }
258        // Ref: https://api.basescan.org/api?module=contract&action=getcontractcreation&contractaddresses=0xC0d3c0d3c0D3c0d3C0D3c0D3C0d3C0D3C0D30010&apiKey=YourAPIKey
259        Err(EtherscanError::Serde { error: _, content }) if content.contains("GENESIS") => {
260            maybe_predeploy = true;
261            Ok((None, maybe_predeploy))
262        }
263        Err(e) => eyre::bail!("Error fetching creation data from verifier-url: {:?}", e),
264    }
265}

267pub fn check_and_encode_args(
268    artifact: &CompactContractBytecode,
269    args: Vec<String>,
270) -> Result<Vec<u8>, eyre::ErrReport> {
271    if let Some(constructor) = artifact.abi.as_ref().and_then(|abi| abi.constructor()) {
272        if constructor.inputs.len() != args.len() {
273            eyre::bail!(
274                "Mismatch of constructor arguments length. Expected {}, got {}",
275                constructor.inputs.len(),
276                args.len()
277            );
278        }
279        encode_args(&constructor.inputs, &args).map(|args| DynSolValue::Tuple(args).abi_encode())
280    } else {
281        Ok(Vec::new())
282    }
283}

293pub fn check_args_len(
294    artifact: &CompactContractBytecode,
295    args: &Bytes,
296) -> Result<(), eyre::ErrReport> {

299            eyre::bail!(
300                "Contract expects {} constructor argument(s), but none were provided",

303        }

308pub async fn get_tracing_executor(
309    fork_config: &mut Config,
310    fork_blk_num: u64,
311    evm_version: EvmVersion,
312    evm_opts: EvmOpts,
313) -> Result<(Env, TracingExecutor)> {
314    fork_config.fork_block_number = Some(fork_blk_num);
315    fork_config.evm_version = evm_version;
316
317    let create2_deployer = evm_opts.create2_deployer;
318    let (env, fork, _chain, is_odyssey) =
319        TracingExecutor::get_fork_material(fork_config, evm_opts).await?;
320
321    let executor = TracingExecutor::new(
322        env.clone(),
323        fork,
324        Some(fork_config.evm_version),
325        TraceMode::Call,
326        is_odyssey,
327        create2_deployer,
328    )?;
329
330    Ok((env, executor))
331}

342pub fn deploy_contract(
343    executor: &mut TracingExecutor,
344    env: &Env,
345    spec_id: SpecId,
346    to: Option<TxKind>,
347) -> Result<Address, eyre::ErrReport> {
348    let env_with_handler = EnvWithHandlerCfg::new(Box::new(env.clone()), HandlerCfg::new(spec_id));
349
350    if to.is_some_and(|to| to.is_call()) {
351        let TxKind::Call(to) = to.unwrap() else { unreachable!() };
352        if to != DEFAULT_CREATE2_DEPLOYER {
353            eyre::bail!("Transaction `to` address is not the default create2 deployer i.e the tx is not a contract creation tx.");
354        }
355        let result = executor.transact_with_env(env_with_handler)?;
356
357        trace!(transact_result = ?result.exit_reason);
358        if result.result.len() != 20 {
359            eyre::bail!(
360                "Failed to deploy contract on fork at block: call result is not exactly 20 bytes"
361            );
362        }
363
364        Ok(Address::from_slice(&result.result))
365    } else {
366        let deploy_result = executor.deploy_with_env(env_with_handler, None)?;
367        trace!(deploy_result = ?deploy_result.raw.exit_reason);
368        Ok(deploy_result.address)
369    }
370}

372pub async fn get_runtime_codes(
373    executor: &mut TracingExecutor,
374    provider: &RetryProvider,
375    address: Address,
376    fork_address: Address,
377    block: Option<u64>,
378) -> Result<(Bytecode, Bytes)> {
379    let fork_runtime_code = executor
380        .backend_mut()
381        .basic(fork_address)?
382        .ok_or_else(|| {
383            eyre::eyre!(
384                "Failed to get runtime code for contract deployed on fork at address {}",
385                fork_address
386            )
387        })?
388        .code
389        .ok_or_else(|| {
390            eyre::eyre!(
391                "Bytecode does not exist for contract deployed on fork at address {}",
392                fork_address
393            )
394        })?;
395
396    let onchain_runtime_code = if let Some(block) = block {
397        provider.get_code_at(address).block_id(BlockId::number(block)).await?
398    } else {
399        provider.get_code_at(address).await?
400    };
401
402    Ok((fork_runtime_code, onchain_runtime_code))
403}
