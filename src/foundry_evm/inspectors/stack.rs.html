167    /// Builds the stack of inspectors to use when transacting/committing on the EVM.
168    pub fn build(self) -> InspectorStack {
169        let Self {
170            block,
171            gas_price,
172            cheatcodes,
173            fuzzer,
174            trace_mode,
175            logs,
176            coverage,
177            print,
178            chisel_state,
179            enable_isolation,
180            odyssey,
181            wallets,
182            create2_deployer,
183        } = self;
184        let mut stack = InspectorStack::new();
185
186        // inspectors
187        if let Some(config) = cheatcodes {
188            let mut cheatcodes = Cheatcodes::new(config);
189            // Set wallets if they are provided
190            if let Some(wallets) = wallets {
191                cheatcodes.set_wallets(wallets);
192            }
193            stack.set_cheatcodes(cheatcodes);
194        }
195
196        if let Some(fuzzer) = fuzzer {
197            stack.set_fuzzer(fuzzer);
198        }
199        if let Some(chisel_state) = chisel_state {
200            stack.set_chisel(chisel_state);
201        }
202
203        stack.collect_coverage(coverage.unwrap_or(false));
204        stack.collect_logs(logs.unwrap_or(true));
205        stack.print(print.unwrap_or(false));
206        stack.tracing(trace_mode);
207
208        stack.enable_isolation(enable_isolation);
209        stack.odyssey(odyssey);
210        stack.set_create2_deployer(create2_deployer);
211
212        // environment, must come after all of the inspectors
213        if let Some(block) = block {
214            stack.set_block(&block);
215        }
216        if let Some(gas_price) = gas_price {
217            stack.set_gas_price(gas_price);
218        }
219
220        stack
221    }
222}

224/// Helper macro to call the same method on multiple inspectors without resorting to dynamic
225/// dispatch.
226#[macro_export]
227macro_rules! call_inspectors {
228    ([$($inspector:expr),+ $(,)?], |$id:ident $(,)?| $call:expr $(,)?) => {
229        $(
230            if let Some($id) = $inspector {
231                ({ #[inline(always)] #[cold] || $call })();
232            }
233        )+
234    };
235    (#[ret] [$($inspector:expr),+ $(,)?], |$id:ident $(,)?| $call:expr $(,)?) => {
236        $(
237            if let Some($id) = $inspector {
238                if let Some(result) = ({ #[inline(always)] #[cold] || $call })() {
239                    return result;
240                }
241            }
242        )+
243    };
244}

449    /// Collects all the data gathered during inspection into a single struct.
450    #[inline]
451    pub fn collect(self) -> InspectorData {
452        let Self {
453            mut cheatcodes,
454            inner: InspectorStackInner { chisel_state, coverage, log_collector, tracer, .. },
455        } = self;
456
457        let traces = tracer.map(|tracer| tracer.into_traces()).map(|arena| {
458            let ignored = cheatcodes
459                .as_mut()
460                .map(|cheatcodes| {
461                    let mut ignored = std::mem::take(&mut cheatcodes.ignored_traces.ignored);
462
463                    // If the last pause call was not resumed, ignore the rest of the trace
464                    if let Some(last_pause_call) = cheatcodes.ignored_traces.last_pause_call {
465                        ignored.insert(last_pause_call, (arena.nodes().len(), 0));
466                    }
467
468                    ignored
469                })
470                .unwrap_or_default();
471
472            SparsedTraceArena { arena, ignored }
473        });
474
475        InspectorData {
476            logs: log_collector.map(|logs| logs.logs).unwrap_or_default(),
477            labels: cheatcodes
478                .as_ref()
479                .map(|cheatcodes| cheatcodes.labels.clone())
480                .unwrap_or_default(),
481            traces,
482            coverage: coverage.map(|coverage| coverage.finish()),
483            cheatcodes,
484            chisel_state: chisel_state.and_then(|state| state.state),
485        }
486    }

505    fn do_call_end(
506        &mut self,
507        ecx: &mut EvmContext<&mut dyn DatabaseExt>,
508        inputs: &CallInputs,
509        outcome: CallOutcome,
510    ) -> CallOutcome {
511        let result = outcome.result.result;
512        call_inspectors!(
513            #[ret]
514            [&mut self.fuzzer, &mut self.tracer, &mut self.cheatcodes, &mut self.printer],
515            |inspector| {
516                let new_outcome = inspector.call_end(ecx, inputs, outcome.clone());
517
518                // If the inspector returns a different status or a revert with a non-empty message,
519                // we assume it wants to tell us something
520                let different = new_outcome.result.result != result ||
521                    (new_outcome.result.result == InstructionResult::Revert &&
522                        new_outcome.output() != outcome.output());
523                different.then_some(new_outcome)
524            },
525        );
526
527        outcome
528    }

580    fn transact_inner(
581        &mut self,
582        ecx: &mut EvmContext<&mut dyn DatabaseExt>,
583        transact_to: TransactTo,
584        caller: Address,
585        input: Bytes,
586        gas_limit: u64,
587        value: U256,
588    ) -> (InterpreterResult, Option<Address>) {
589        let ecx = &mut ecx.inner;
590
591        let cached_env = ecx.env.clone();
592
593        ecx.env.block.basefee = U256::ZERO;
594        ecx.env.tx.caller = caller;
595        ecx.env.tx.transact_to = transact_to;
596        ecx.env.tx.data = input;
597        ecx.env.tx.value = value;
598        // Add 21000 to the gas limit to account for the base cost of transaction.
599        ecx.env.tx.gas_limit = gas_limit + 21000;
600        // If we haven't disabled gas limit checks, ensure that transaction gas limit will not
601        // exceed block gas limit.
602        if !ecx.env.cfg.disable_block_gas_limit {
603            ecx.env.tx.gas_limit =
604                std::cmp::min(ecx.env.tx.gas_limit, ecx.env.block.gas_limit.to());
605        }
606        ecx.env.tx.gas_price = U256::ZERO;
607
608        self.inner_context_data = Some(InnerContextData { original_origin: cached_env.tx.caller });
609        self.in_inner_context = true;
610
611        let env = EnvWithHandlerCfg::new_with_spec_id(ecx.env.clone(), ecx.spec_id());
612        let res = self.with_stack(|inspector| {
613            let mut evm = crate::utils::new_evm_with_inspector(&mut ecx.db, env, inspector);
614
615            evm.context.evm.inner.journaled_state.state = {
616                let mut state = ecx.journaled_state.state.clone();
617
618                for (addr, acc_mut) in &mut state {
619                    // mark all accounts cold, besides preloaded addresses
620                    if !ecx.journaled_state.warm_preloaded_addresses.contains(addr) {
621                        acc_mut.mark_cold();
622                    }
623
624                    // mark all slots cold
625                    for slot_mut in acc_mut.storage.values_mut() {
626                        slot_mut.is_cold = true;
627                        slot_mut.original_value = slot_mut.present_value;
628                    }
629                }
630
631                state
632            };
633
634            // set depth to 1 to make sure traces are collected correctly
635            evm.context.evm.inner.journaled_state.depth = 1;
636
637            let res = evm.transact();
638
639            // need to reset the env in case it was modified via cheatcodes during execution
640            ecx.env = evm.context.evm.inner.env;
641            res
642        });
643
644        self.in_inner_context = false;
645        self.inner_context_data = None;
646
647        ecx.env.tx = cached_env.tx;
648        ecx.env.block.basefee = cached_env.block.basefee;
649
650        let mut gas = Gas::new(gas_limit);
651
652        let Ok(res) = res else {
653            // Should we match, encode and propagate error as a revert reason?
654            let result =
655                InterpreterResult { result: InstructionResult::Revert, output: Bytes::new(), gas };
656            return (result, None);
657        };
658
659        for (addr, mut acc) in res.state {
660            let Some(acc_mut) = ecx.journaled_state.state.get_mut(&addr) else {
661                ecx.journaled_state.state.insert(addr, acc);
662                continue
663            };
664
665            // make sure accounts that were warmed earlier do not become cold
666            if acc.status.contains(AccountStatus::Cold) &&
667                !acc_mut.status.contains(AccountStatus::Cold)
668            {
669                acc.status -= AccountStatus::Cold;
670            }
671            acc_mut.info = acc.info;
672            acc_mut.status |= acc.status;
673
674            for (key, val) in acc.storage {
675                let Some(slot_mut) = acc_mut.storage.get_mut(&key) else {
676                    acc_mut.storage.insert(key, val);
677                    continue
678                };
679                slot_mut.present_value = val.present_value;
680                slot_mut.is_cold &= val.is_cold;
681            }
682        }
683
684        let (result, address, output) = match res.result {
685            ExecutionResult::Success { reason, gas_used, gas_refunded, logs: _, output } => {
686                gas.set_refund(gas_refunded as i64);
687                let _ = gas.record_cost(gas_used);
688                let address = match output {
689                    Output::Create(_, address) => address,
690                    Output::Call(_) => None,
691                };
692                (reason.into(), address, output.into_data())
693            }
694            ExecutionResult::Halt { reason, gas_used } => {
695                let _ = gas.record_cost(gas_used);
696                (reason.into(), None, Bytes::new())
697            }
698            ExecutionResult::Revert { gas_used, output } => {
699                let _ = gas.record_cost(gas_used);
700                (InstructionResult::Revert, None, output)
701            }
702        };
703        (InterpreterResult { result, output, gas }, address)
704    }
705
706    /// Moves out of references, constructs an [`InspectorStack`] and runs the given closure with
707    /// it.
708    fn with_stack<O>(&mut self, f: impl FnOnce(&mut InspectorStack) -> O) -> O {
709        let mut stack = InspectorStack {
710            cheatcodes: self
711                .cheatcodes
712                .as_deref_mut()
713                .map(|cheats| core::mem::replace(cheats, Cheatcodes::new(cheats.config.clone()))),
714            inner: std::mem::take(self.inner),
715        };
716
717        let out = f(&mut stack);
718
719        if let Some(cheats) = self.cheatcodes.as_deref_mut() {
720            *cheats = stack.cheatcodes.take().unwrap();
721        }
722
723        *self.inner = stack.inner;
724
725        out
726    }
727
728    /// Invoked at the beginning of a new top-level (0 depth) frame.
729    fn top_level_frame_start(&mut self, ecx: &mut EvmContext<&mut dyn DatabaseExt>) {
730        if self.enable_isolation {
731            // If we're in isolation mode, we need to keep track of the state at the beginning of
732            // the frame to be able to roll back on revert
733            self.top_frame_journal = ecx.journaled_state.state.clone();
734        }
735    }
736
737    /// Invoked at the end of root frame.
738    fn top_level_frame_end(
739        &mut self,
740        ecx: &mut EvmContext<&mut dyn DatabaseExt>,
741        result: InstructionResult,
742    ) {
743        if !result.is_revert() {
744            return;
745        }
746        // Encountered a revert, since cheatcodes may have altered the evm state in such a way
747        // that violates some constraints, e.g. `deal`, we need to manually roll back on revert
748        // before revm reverts the state itself
749        if let Some(cheats) = self.cheatcodes.as_mut() {
750            cheats.on_revert(ecx);
751        }
752
753        // If we're in isolation mode, we need to rollback to state before the root frame was
754        // created We can't rely on revm's journal because it doesn't account for changes
755        // made by isolated calls
756        if self.enable_isolation {
757            ecx.journaled_state.state = std::mem::take(&mut self.top_frame_journal);
758        }
759    }
760}

817    fn call(
818        &mut self,
819        ecx: &mut EvmContext<&mut dyn DatabaseExt>,
820        call: &mut CallInputs,
821    ) -> Option<CallOutcome> {
822        if self.in_inner_context && ecx.journaled_state.depth == 1 {
823            self.adjust_evm_data_for_inner_context(ecx);
824            return None;
825        }
826
827        if ecx.journaled_state.depth == 0 {
828            self.top_level_frame_start(ecx);
829        }
830
831        call_inspectors!(
832            #[ret]
833            [&mut self.fuzzer, &mut self.tracer, &mut self.log_collector, &mut self.printer],
834            |inspector| {
835                let mut out = None;
836                if let Some(output) = inspector.call(ecx, call) {
837                    if output.result.result != InstructionResult::Continue {
838                        out = Some(Some(output));
839                    }
840                }
841                out
842            },
843        );
844
845        if let Some(cheatcodes) = self.cheatcodes.as_deref_mut() {
846            // Handle mocked functions, replace bytecode address with mock if matched.
847            if let Some(mocks) = cheatcodes.mocked_functions.get(&call.target_address) {
848                // Check if any mock function set for call data or if catch-all mock function set
849                // for selector.
850                if let Some(target) = mocks
851                    .get(&call.input)
852                    .or_else(|| call.input.get(..4).and_then(|selector| mocks.get(selector)))
853                {
854                    call.bytecode_address = *target;
855                }
856            }
857
858            if let Some(output) = cheatcodes.call_with_executor(ecx, call, self.inner) {
859                if output.result.result != InstructionResult::Continue {
860                    return Some(output);
861                }
862            }
863        }
864
865        if self.enable_isolation && !self.in_inner_context && ecx.journaled_state.depth == 1 {
866            match call.scheme {
867                // Isolate CALLs
868                CallScheme::Call | CallScheme::ExtCall => {
869                    let (result, _) = self.transact_inner(
870                        ecx,
871                        TxKind::Call(call.target_address),
872                        call.caller,
873                        call.input.clone(),
874                        call.gas_limit,
875                        call.value.get(),
876                    );
877                    return Some(CallOutcome {
878                        result,
879                        memory_offset: call.return_memory_offset.clone(),
880                    });
881                }
882                // Mark accounts and storage cold before STATICCALLs
883                CallScheme::StaticCall | CallScheme::ExtStaticCall => {
884                    let JournaledState { state, warm_preloaded_addresses, .. } =
885                        &mut ecx.journaled_state;
886                    for (addr, acc_mut) in state {
887                        // Do not mark accounts and storage cold accounts with arbitrary storage.
888                        if let Some(cheatcodes) = &self.cheatcodes {
889                            if cheatcodes.has_arbitrary_storage(addr) {
890                                continue;
891                            }
892                        }
893
894                        if !warm_preloaded_addresses.contains(addr) {
895                            acc_mut.mark_cold();
896                        }
897
898                        for slot_mut in acc_mut.storage.values_mut() {
899                            slot_mut.is_cold = true;
900                        }
901                    }
902                }
903                // Process other variants as usual
904                CallScheme::CallCode | CallScheme::DelegateCall | CallScheme::ExtDelegateCall => {}
905            }
906        }
907
908        None
909    }
910
911    fn call_end(
912        &mut self,
913        ecx: &mut EvmContext<&mut dyn DatabaseExt>,
914        inputs: &CallInputs,
915        outcome: CallOutcome,
916    ) -> CallOutcome {
917        // We are processing inner context outputs in the outer context, so need to avoid processing
918        // twice.
919        if self.in_inner_context && ecx.journaled_state.depth == 1 {
920            return outcome;
921        }
922
923        let outcome = self.do_call_end(ecx, inputs, outcome);
924
925        if ecx.journaled_state.depth == 0 {
926            self.top_level_frame_end(ecx, outcome.result.result);
927        }
928
929        outcome
930    }
