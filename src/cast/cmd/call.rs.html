35// matches override pattern <address>:<slot>:<value>
36// e.g. 0x123:0x1:0x1234
37static OVERRIDE_PATTERN: LazyLock<Regex> =
38    LazyLock::new(|| Regex::new(r"^([^:]+):([^:]+):([^:]+)$").unwrap());

40/// CLI arguments for `cast call`.
41///
42/// ## State Override Flags
43///
44/// The following flags can be used to override the state for the call:
45///
46/// * `--override-balance <address>:<balance>` - Override the balance of an account
47/// * `--override-nonce <address>:<nonce>` - Override the nonce of an account
48/// * `--override-code <address>:<code>` - Override the code of an account
49/// * `--override-state <address>:<slot>:<value>` - Override a storage slot of an account
50///
51/// Multiple overrides can be specified for the same account. For example:
52///
53/// ```bash
54/// cast call 0x... "transfer(address,uint256)" 0x... 100 \
55///   --override-balance 0x123:0x1234 \
56///   --override-nonce 0x123:1 \
57///   --override-code 0x123:0x1234 \
58///   --override-state 0x123:0x1:0x1234
59///   --override-state-diff 0x123:0x1:0x1234
60/// ```
61#[derive(Debug, Parser)]

175impl CallArgs {
176    pub async fn run(self) -> Result<()> {
177        let figment = Into::<Figment>::into(&self.eth).merge(&self);
178        let evm_opts = figment.extract::<EvmOpts>()?;
179        let mut config = Config::from_provider(figment)?.sanitized();
180        let state_overrides = self.get_state_overrides()?;
181
182        let Self {
183            to,
184            mut sig,
185            mut args,
186            mut tx,
187            eth,
188            command,
189            block,
190            trace,
191            evm_version,
192            debug,
193            decode_internal,
194            labels,
195            data,
196            with_local_artifacts,
197            ..
198        } = self;
199
200        if let Some(data) = data {
201            sig = Some(data);
202        }
203
204        let provider = utils::get_provider(&config)?;
205        let sender = SenderKind::from_wallet_opts(eth.wallet).await?;
206        let from = sender.address();
207
208        let code = if let Some(CallSubcommands::Create {
209            code,
210            sig: create_sig,
211            args: create_args,
212            value,
213        }) = command
214        {
215            sig = create_sig;
216            args = create_args;
217            if let Some(value) = value {
218                tx.value = Some(value);
219            }
220            Some(code)
221        } else {
222            None
223        };
224
225        let (tx, func) = CastTxBuilder::new(&provider, tx, &config)
226            .await?
227            .with_to(to)
228            .await?
229            .with_code_sig_and_args(code, sig, args)
230            .await?
231            .build_raw(sender)
232            .await?;
233
234        if trace {
235            if let Some(BlockId::Number(BlockNumberOrTag::Number(block_number))) = self.block {
236                // Override Config `fork_block_number` (if set) with CLI value.
237                config.fork_block_number = Some(block_number);
238            }
239
240            let create2_deployer = evm_opts.create2_deployer;
241            let (mut env, fork, chain, odyssey) =
242                TracingExecutor::get_fork_material(&config, evm_opts).await?;
243
244            // modify settings that usually set in eth_call
245            env.cfg.disable_block_gas_limit = true;
246            env.block.gas_limit = U256::MAX;
247
248            let trace_mode = TraceMode::Call
249                .with_debug(debug)
250                .with_decode_internal(if decode_internal {
251                    InternalTraceMode::Full
252                } else {
253                    InternalTraceMode::None
254                })
255                .with_state_changes(shell::verbosity() > 4);
256            let mut executor = TracingExecutor::new(
257                env,
258                fork,
259                evm_version,
260                trace_mode,
261                odyssey,
262                create2_deployer,
263            )?;
264
265            let value = tx.value.unwrap_or_default();
266            let input = tx.inner.input.into_input().unwrap_or_default();
267            let tx_kind = tx.inner.to.expect("set by builder");
268
269            if let Some(access_list) = tx.inner.access_list {
270                executor.env_mut().tx.access_list = access_list.0
271            }
272
273            let trace = match tx_kind {
274                TxKind::Create => {
275                    let deploy_result = executor.deploy(from, input, value, None);
276                    TraceResult::try_from(deploy_result)?
277                }
278                TxKind::Call(to) => TraceResult::from_raw(
279                    executor.transact_raw(from, to, input, value)?,
280                    TraceKind::Execution,
281                ),
282            };
283
284            handle_traces(
285                trace,
286                &config,
287                chain,
288                labels,
289                with_local_artifacts,
290                debug,
291                decode_internal,
292            )
293            .await?;
294
295            return Ok(());
296        }
297
298        sh_println!(
299            "{}",
300            Cast::new(provider).call(&tx, func.as_ref(), block, state_overrides).await?
301        )?;
302
303        Ok(())
304    }
305    /// Parse state overrides from command line arguments
306    pub fn get_state_overrides(&self) -> eyre::Result<StateOverride> {
307        let mut state_overrides_builder = StateOverridesBuilder::default();
308
309        // Parse balance overrides
310        for override_str in self.balance_overrides.iter().flatten() {
311            let (addr, balance) = address_value_override(override_str)?;
312            state_overrides_builder =
313                state_overrides_builder.with_balance(addr.parse()?, balance.parse()?);
314        }
315
316        // Parse nonce overrides
317        for override_str in self.nonce_overrides.iter().flatten() {
318            let (addr, nonce) = address_value_override(override_str)?;
319            state_overrides_builder =
320                state_overrides_builder.with_nonce(addr.parse()?, nonce.parse()?);
321        }
322
323        // Parse code overrides
324        for override_str in self.code_overrides.iter().flatten() {
325            let (addr, code_str) = address_value_override(override_str)?;
326            state_overrides_builder =
327                state_overrides_builder.with_code(addr.parse()?, Bytes::from_str(code_str)?);
328        }
329
330        // Parse state overrides
331        for override_str in self.state_overrides.iter().flatten() {
332            let (addr, slot, value) = address_slot_value_override(override_str)?;
333            state_overrides_builder =
334                state_overrides_builder.with_state(addr, [(slot.into(), value.into())]);
335        }
336
337        // Parse state diff overrides
338        for override_str in self.state_diff_overrides.iter().flatten() {
339            let (addr, slot, value) = address_slot_value_override(override_str)?;
340            state_overrides_builder =
341                state_overrides_builder.with_state_diff(addr, [(slot.into(), value.into())]);
342        }
343
344        Ok(state_overrides_builder.build())
345    }
346}

368/// Parse an override string in the format address:value.
369fn address_value_override(address_override: &str) -> Result<(&str, &str)> {
370    address_override.split_once(':').ok_or_else(|| {
371        eyre::eyre!("Invalid override {address_override}. Expected <address>:<value>")
372    })
373}
374
375/// Parse an override string in the format address:slot:value.
376fn address_slot_value_override(address_override: &str) -> Result<(Address, U256, U256)> {
377    let captures = OVERRIDE_PATTERN.captures(address_override).ok_or_else(|| {
378        eyre::eyre!("Invalid override {address_override}. Expected <address>:<slot>:<value>")
379    })?;
380
381    Ok((
382        captures[1].parse()?, // Address
383        captures[2].parse()?, // Slot (U256)
384        captures[3].parse()?, // Value (U256)
385    ))
386}
